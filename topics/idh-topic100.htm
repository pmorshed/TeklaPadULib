using System;
using Tekla.Structures.Model;
using Tekla.Structures.Geometry3d;

class MyClass
{
          static void Main()
          {
                    try
                    {
                              Model m1 = new Model();
                              Point p1 = new Point(0, 0, 0);
                              Point p2 = new Point(1000, 2000, 6000);
                              Beam b1 = new Beam();
                              b1.StartPoint = p1;
                              b1.EndPoint = p2;
                              b1.Profile.ProfileString = "IPE200";
                              b1.Material.MaterialString = "S235JR";
                              b1.Insert();
                              m1.CommitChanges();
                    }
                    catch (Exception)
                    { }
          }
}
