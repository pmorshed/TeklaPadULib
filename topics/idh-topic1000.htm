//## Description: Swapping handles for all selected beams
//## Compatibility: TS18 or newer


//A very basic example to start:

Model m1 = new Model();
//Walking through beam/columns inside database (selection):
foreach (var item in Database.OfType<Beam>())
{
    //Getting current handles:
    var SPoint = item.StartPoint;
    var EPoint = item.EndPoint;
    //Swapping handles:
    item.StartPoint = EPoint;
    item.EndPoint = SPoint;
    //Modifying:
    item.Modify();
}
//Updating the model view:
m1.CommitChanges();
//Passing nothing to data grid:
return null;
