//## Description: Cast-unit by GUID
//## Measurement units: Metric
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering "Assemblies":
var query = from n in Database.OfType<Assembly>()
            //Defining required columns:
            let CU_NAME = n.GetTeklaReport("cast unit name")
            let Pos = n.GetTeklaReport("cast unit pos")
            let Material = n.GetTeklaReport("material")
            let RebarWeight = n.GetTeklaReport("cast unit rebar weight", 4)
            let CU_Type = n.GetTeklaReport("cast unit type")
            //Converting to cubic meter and rounding after conversion:
            let Volume = Math.Round(n.GetTeklaReport("volume net", 9) / 1000000000, 3)
            //Converting to Ton:
            let Weight = n.GetTeklaReport("weight", 4) / 1000
            let location = n.GetTeklaReport("cast unit position code")
            let Phase = n.GetTeklaReport("phase")
            select new { CU_NAME, Material, Pos, location, CU_Type, Phase, RebarWeight, Volume, Weight, n.Identifier.GUID };
//Passing the final result to the data grid:
return query;
