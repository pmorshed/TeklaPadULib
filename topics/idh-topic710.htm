//Walking through database items(selected items) and filtering "Parts":
var result = from n in Database.OfType<Part>()
//Walking through bolt groups in each part:        
from BoltGroup b in n.GetBolts().GetArray()
//Gerouping bolts with size and owner:
let owner = n.Profile.ProfileString + "[" + n.Identifier.ID + "]"
let BoltSize = b.BoltSize
group b by new { owner, BoltSize } into g
//Defining required columns:
let No = g.Sum(x => x.GetTeklaReport("number"))
//Selecting a new table (from orginal database) with required columns:
select new { g.Key.owner, bolt = "Bolt-"+g.Key.BoltSize, No };
//Passing the final result to the gridview:
return result;
