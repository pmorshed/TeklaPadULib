//## Description: Pour list by cast-unit
//## Measurement units: Metric
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering pour objects:
var result = from n in Database.OfType<PourObject>()
                 //Defining required columns:
             let PourNumber = n.GetTeklaReport("pour number")
             let PourType = n.GetTeklaReport("pour type")
             let Mixture = n.GetTeklaReport("concrete mixture")
             let Volume_m3 = Math.Round(n.GetTeklaReport("volume", 9) / 1000000000, 3)
             let Weight = n.GetTeklaReport("Weight")
             let CastUnitID = "CastUnit-" + n.GetParts().GetArray().OfType<Part>().First().Identifier.ID
             select new { CastUnitID, PourNumber, PourType, Mixture, Volume_m3, Weight };
//Passing the final result to the data grid:
return result;
