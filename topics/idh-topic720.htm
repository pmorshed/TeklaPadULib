//Walking through database items(selected items) and filtering "Assemblies":
var RebarStrands = from p in Database.OfType<Part>()
//Filtering cast units:
where p.GetTeklaReport("material type") == "CONCRETE"
from Reinforcement r in p.GetReinforcements().GetArray()
where r.GetType().Name == "RebarStrand" && r.IsGeometryValid()
//Defining required columns:
let PartID= p.Identifier.ID
let PartPos = r.GetTeklaReport("cast unit pos")
let Pos = r.GetTeklaReport("rebar pos")
let Size = r.GetTeklaReport("size")
let grade = r.GetTeklaReport("grade")
let force = r.GetTeklaReport("strand pull force")
let length = r.GetTeklaReport("length")
let number = r.GetTeklaReport("number")
//Selecting a new table (from orginal database) with required columns:
select new {PartID, PartPos, Pos, Size, grade, force, length, number };
//Passing the final result to the gridview:	        
return RebarStrands;
