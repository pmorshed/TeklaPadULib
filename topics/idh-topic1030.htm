//## Description: Removing some lines (in Drawing Database) based on color.
//## Compatibility: TS18 or newer
//## Hint: Make sure that your drawing is open.


var MyDrawing = new TSD.DrawingHandler().GetActiveDrawing();
foreach (var item in DrawingDatabase.OfType<TSD.Line>())
{
    //Condition:
    if (item.Attributes.Line.Color == TSD.DrawingColors.Red)
    {
        item.Delete();
    }
}
//Updating the drawing view:
MyDrawing.CommitChanges();
//Passing nothing to the data grid:
return null;
