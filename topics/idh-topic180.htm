// ### Measurement units: Metric

//Walking through database items(selected items) and filtering parts:
var query = from n in Database.OfType<Part>()
//Finding groups with same profile and same length and same phase:
let key1 = n.Profile.ProfileString
let key2 = n.GetTeklaReport("length net", 3) //precision:3
let key3 = n.GetTeklaReport("phase")
group n by new { key1, key2, key3 } into g
//Defining required columns:
let Profile = g.Key.key1
let length = Math.Round(g.Key.key2 / 1000, 3) //Converting to "meter"
let PhaseNumber = g.Key.key3
let No = g.Count()
//Rounding after summations:
let LengthTotal = Math.Round(length*No, 3) 
let WeightTotal = Math.Round(g.Sum(x => (double)x.GetTeklaReport("weight net")), 2)
//Selecting a new table (from orginal database) with required columns: 
select new { PhaseNumber, Profile, length, No, LengthTotal, WeightTotal };
//Passing the final result to the gridview:
return query;
