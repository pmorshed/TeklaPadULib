//## Author: TeklaPad
//## Description: Plate list by thickness (by phase)
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering parts:
var result = from n in Database.OfType<Part>()
             //Filtering plates (plate profiles and contour plates):
             where n.GetTeklaReport("profile type") == "B"
             //Finding thickness (minimum of dimensions):
             let length = n.GetTeklaReport("length net")
             let width = n.GetTeklaReport("width")
             let height = n.GetTeklaReport("height")
             let thickness = Math.Min(Math.Min(length, width), height)
             let phase = n.GetTeklaReport("phase")
             //Grouping by thickness and phase:
             group n by new { thickness, phase } into g
             //Defining required column:
             let PlatePhase = g.Key.phase
             let WeightTotal = Math.Round(g.Sum(x => (double)x.GetTeklaReport("weight net")), 2)
             select new { PlatePhase, Plate = "Plate" + g.Key.thickness, WeightTotal };
//Passing the final result to the data grid:
return result;
