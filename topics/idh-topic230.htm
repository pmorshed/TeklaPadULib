//Walking through database items(selected items) and filtering "Parts":
var result = from n in Database.OfType<Part>()
//Filtering plates (plate profiles and contour plates): 
where n.GetTeklaReport("profile type") == "B"
//Grouping by position number:
let Pos = n.GetTeklaReport("part pos")
group n by Pos into g
//Defining required column:
let PositionMark = g.Key
let Profile = g.First().Profile.ProfileString
let No = g.Count()
let Grade = g.First().Material.MaterialString
let UnitWeight = g.First().GetTeklaReport("weight net")
let WeightTotal = Math.Round(g.Sum(x => (double)x.GetTeklaReport("weight net")), 2)
//Hint>> Another method(because parts with same position number will have same weight):
//WeightTotal = Math.Round(UnitWeight*No, 2)

//Selecting a new table (from orginal database) with required columns:
select new { PositionMark, Profile, No, Grade, WeightTotal };
//Passing the final result to the gridview:
return result;
