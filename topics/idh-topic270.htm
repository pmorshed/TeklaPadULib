//## Description: Assembly list by pos. number (by phase)
//## Compatibility: TS18 or newer
//## Hint: Make sure that numbering of objects is up to date.


//Walking through database items(selected items) and filtering "Assemblies":
var query = from n in Database.OfType<Assembly>()
            //Finding groups with same pos and same phase:
            let key1 = n.GetTeklaReport("assembly pos")
            let key2 = n.GetTeklaReport("phase")
            group n by new { key1, key2 } into g
            //Keeping one instance (first one) from each group of data:
            let assembly = g.First()
            //Defining required columns:
            let No = g.Count()
            let MP = assembly.GetMainPart() as Part
            let MainPart = MP.Profile.ProfileString
            let Pos = g.Key.key1
            let PhaseNumber = g.Key.key2
            let Weight = assembly.GetTeklaReport("weight net", 3) //Precision:3
            let Name = assembly.Name
            let WeightTotal = Weight * No
            select new { PhaseNumber, Pos, Name, No, MainPart, Weight, WeightTotal };
//Passing the final result to the data grid:
return query;
