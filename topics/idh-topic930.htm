//## Description: PolyBeam exploder.
//## Compatibility: TS18 or newer


Model m1 = new Model();
foreach (var item in Database.OfType<PolyBeam>())
{
    var nodes = item.GetContourPointList();
    for (int i = 0; i < nodes.Count() - 1; i++)
    {
        Point Start = new Point(nodes[i].X, nodes[i].Y, nodes[i].Z);
        Point End = new Point(nodes[i + 1].X, nodes[i + 1].Y, nodes[i + 1].Z);
        Beam b1 = new Beam(Start, End);
        b1.Profile = item.Profile;
        b1.Material = item.Material;
        b1.Class = item.Class;
        b1.Name = item.Name;
        b1.Position = item.Position;
        b1.Insert();
    }
    //Removing original polybeam:
    item.Delete();
}
//Updating the model view:
m1.CommitChanges();
//Passing nothing to data grid:
return null;
