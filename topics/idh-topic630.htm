using System;
using Tekla.Structures.Model;
using Tekla.Structures.Geometry3d;

class MyClass
{
          static void Main()
          {
                    string Prompt1 = "Pick the Primary Part";
                    string Prompt2 = "Pick the Secondary Parts, then Press the Middle Mouse Button When Done";
                    Model Model1 = new Model();
                    Tekla.Structures.Model.UI.Picker Pick1 = new Tekla.Structures.Model.UI.Picker();
                    bool keepgoing = true;
                    while (keepgoing)
                    {
                              try
                              {
                                        Beam Beam1 = Pick1.PickObject(Tekla.Structures.Model.UI.Picker.PickObjectEnum.PICK_ONE_PART, Prompt1) as Beam;
                                        double Xdist = Beam1.StartPoint.X - Beam1.EndPoint.X;
                                        double Ydist = Beam1.StartPoint.Y - Beam1.EndPoint.Y;
                                        Vector Vector1 = new Vector(Xdist, Ydist, 0);

                                        GeometricPlane Plane1 = new GeometricPlane(Beam1.StartPoint, Vector1, new Vector(0, 0, 1));

                                        ModelObjectEnumerator Beams = Pick1.PickObjects(Tekla.Structures.Model.UI.Picker.PickObjectsEnum.PICK_N_PARTS, Prompt2);

                                        while (Beams.MoveNext())
                                        {
                                                  Beam Beam2 = Beams.Current as Beam;
                                                  if (Beam2 != null)
                                                  {

                                                            Point IntersectPoint = Intersection.LineToPlane(new Line(Beam2.StartPoint, Beam2.EndPoint), Plane1);
                                                            Line HelperLine = new Line(IntersectPoint, new Point(IntersectPoint.X, IntersectPoint.Y, IntersectPoint.Z + 100));
                                                            LineSegment Lineseg1 = Intersection.LineToLine(HelperLine, new Line(Beam1.StartPoint, Beam1.EndPoint));
                                                            var D1 = Distance.PointToPoint(Beam2.StartPoint, Lineseg1.Point1);
                                                            var D2 = Distance.PointToPoint(Beam2.EndPoint, Lineseg1.Point1);
                                                            if (D1 < D2)
                                                            {
                                                                      Beam2.StartPoint = Lineseg1.Point1;
                                                                      Beam2.Modify();
                                                            }
                                                            if (D2 < D1)
                                                            {
                                                                      Beam2.EndPoint = Lineseg1.Point1;
                                                                      Beam2.Modify();
                                                            }
                                                  }
                                        }
                                        Model1.CommitChanges();
                              }
                              catch
                              {
                                        keepgoing = false;
                              }
                    }
          }
}
