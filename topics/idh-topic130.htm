//## Description: Distinct assembly list by phase
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering "Assemblies":
var query = from n in Database.OfType<Assembly>()
            //Defining required columns:
            let Name = n.Name
            let MP = n.GetMainPart() as Part
            let MainPart = MP.Profile.ProfileString
            let Pos = n.GetTeklaReport("assembly pos")
            let Weight = n.GetTeklaReport("weight net", 3)
            let AssemblyPhase = n.GetTeklaReport("phase")
            let SecondaryParts = n.GetSecondaries().Count
            let SubAssemblies = n.GetSubAssemblies().Count
            select new { Pos, MainPart, Weight, AssemblyPhase, SubAssemblies, SecondaryParts };
//Passing the final result to the data grid:
return query;
