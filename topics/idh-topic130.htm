//Walking through database items(selected items):
var query = from n in Database
//Filtering "Assemblies" from the database:
let a = n as Assembly
where a != null
//Removing cast units:
where a.GetTeklaReport("material type") == "STEEL"
//Defining required columns:
let Name = a.Name
let MP = a.GetMainPart() as Part
let MainPart = MP.Profile.ProfileString
let Pos = a.GetTeklaReport("assembly pos")
let Weight = a.GetTeklaReport("weight net",3)
let AssemblyPhase = a.GetTeklaReport("phase")
let SecondaryParts = a.GetSecondaries().Count
let SubAssemblies = a.GetSubAssemblies().Count
//Selecting a new table (from orginal database) with required columns:
select new { Pos, MainPart, Weight, AssemblyPhase, SubAssemblies, SecondaryParts };
//Passing the final result to the gridview:
return query;
