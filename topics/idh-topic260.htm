//Walking through database items(selected items) and filtering "Assemblies":
var query = from n in Database.OfType<Assembly>()
//Removing cast units:
where n.GetTeklaReport("material type") == "STEEL"
//Grouping by phase number:
group n by n.GetTeklaReport("phase") into g
//Defining required columns:
let AssemblyPhase = g.Key
let WeightTotal = g.Sum(x=> (double)x.GetTeklaReport("weight net"))
//Selecting a new table (from orginal database) with required columns:
select new { AssemblyPhase, WeightTotal };
//Passing the final result to the gridview:
return query;
