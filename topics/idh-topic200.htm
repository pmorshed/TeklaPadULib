//## Description: Beam coordination list
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering "Beams":
var query = from n in Database.OfType<Beam>()
            //Filtering properties (removing strip footings,...):
            where (n.Type == Beam.BeamTypeEnum.BEAM) || (n.Type == Beam.BeamTypeEnum.COLUMN)
            //Defining required columns:
            let Start_X = n.StartPoint.X
            let Start_Y = n.StartPoint.Y
            let StartZ = n.StartPoint.Z
            let End_X = n.EndPoint.X
            let End_Y = n.EndPoint.Y
            let End_Z = n.EndPoint.Z
            let PartPos = n.GetTeklaReport("part pos")
            select new { PartPos, Start_X, Start_Y, StartZ, End_X, End_Y, End_Z };
//Passing the final result to the data grid:
return query;
