//Walking through database items(selected items) and filtering "Parts":
var query = from n in Database.OfType<Part>()
//Filtering Beams and PolyBeams:
where n is Beam || n is PolyBeam 
//Hint>> Another method:
//where n.GetType().Name=="Beam" || n.GetType().Name=="PolyBeam"

//Defining required columns:
let Start_X = n.GetTeklaReport("start x")
let Start_Y = n.GetTeklaReport("start y")
let StartZ = n.GetTeklaReport("start z")
let End_X= n.GetTeklaReport("end x")
let End_Y = n.GetTeklaReport("end y")
let End_Z= n.GetTeklaReport("end z")
let PartType= n.GetType().Name
let PartPos = n.GetTeklaReport("part pos")

//Selecting a new table (from orginal database) with required columns: 
select new {PartPos, PartType, Start_X, Start_Y,StartZ, End_X, End_Y, End_Z};

//Passing the final result to the gridview:
return query;
