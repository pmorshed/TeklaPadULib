//## Description: Selecting similar assemblies
//## Compatibility: TS18 or newer

//## Hint: Make sure that numbering of objects is up to date

Model m1 = new Model();
//Let user to pick one assembly in model:
var RefAssembly = m1.PickObjectOfType<Assembly>();

//Finding similar assemblies:
var SimilarAssemblies = (from n in OfType<Assembly>()
                             //Similarity conditions:
                         where n.GetTeklaReport("assembly pos") == RefAssembly.GetTeklaReport("assembly pos")
                         select n).Highlight();

//Passing nothing to data grid:
return null;
