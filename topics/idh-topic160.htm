//## Description: Profile list by length
//## Measurement units: Metric
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering parts:
var query = from n in Database.OfType<Part>()
            //Finding data groups with same profile and same length:
            let key1 = n.Profile.ProfileString
            let key2 = n.GetTeklaReport("length net")
            group n by new { key1, key2 } into g
            //Defining required columns:
            let Profile = g.Key.key1
            let length = g.Key.key2 / 1000 //Converting to "meter"
            let No = g.Count()
            let LengthTotal = length * No
            //Rounding after summation:
            let WeightTotal = Math.Round(g.Sum(x => (double)x.GetTeklaReport("weight net")), 2)
            select new { Profile, length, No, LengthTotal, WeightTotal };
//Passing the final result to the data grid:
return query;
