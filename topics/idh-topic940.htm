//## Description: Complex label for beams
//## Compatibility: TS18 or newer


TSMUI.GraphicsDrawer drawer = new TSMUI.GraphicsDrawer();
Model m1 = new Model();
foreach (var item in Database.OfType<Beam>())
{
    //Finding midpoint:
    var x_mid = (item.StartPoint.X + item.EndPoint.X) / 2;
    var y_mid = (item.StartPoint.Y + item.EndPoint.Y) / 2;
    var z_mid = (item.StartPoint.Z + item.EndPoint.Z) / 2;
    var MidPoint = new Point(x_mid, y_mid, z_mid);
    //Defining the needed properties:
    var top = "Elv.:" + item.GetTeklaReport("top level");
    var weight = "weight is:" + item.GetTeklaReport("weight net");
    var profile = item.Profile.ProfileString;
    var phase = "Phase:" + item.GetTeklaReport("phase");
    //Creating label:
    drawer.DrawText(new Point(x_mid, y_mid, z_mid + 800), profile, new TSMUI.Color(0, 0, 0));
    drawer.DrawText(new Point(x_mid, y_mid, z_mid + 600), top, new TSMUI.Color(1, 1, 0));
    drawer.DrawText(new Point(x_mid, y_mid, z_mid + 400), phase, new TSMUI.Color(1, 0, 0));
    drawer.DrawText(new Point(x_mid, y_mid, z_mid + 200), weight, new TSMUI.Color(0, 0, 1));
    drawer.DrawLineSegment(MidPoint, new Point(x_mid, y_mid, z_mid + 900), new TSMUI.Color(1, 1, 1));
}
//Updating the model view:
m1.CommitChanges();
//Passing nothing to data grid:
return null;
