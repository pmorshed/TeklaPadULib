// ### Compatibility: TS18 or newer

//Gathering data for Nut1:
var nut1 = from n in Database.OfType<BoltGroup>()
//Making sure that nut1 exists in boltgroup:
where n.Bolt == true && n.Nut1 == true && n.GetTeklaReport("nut.weight1") > 0
//Grouping by name and phase:
group n by new { key1 = n.GetTeklaReport("nut.name1"), key2 = n.GetTeklaReport("phase") } into g
//Defining columns:
let Name = g.Key.key1
let InnerD = g.First().GetTeklaReport("nut.inner diameter1")
let Number = g.Sum(x => x.GetTeklaReport("nut.number1"))
let Weight = g.Sum(x => (double)(x.GetTeklaReport("nut.weight1", 4)*x.GetTeklaReport("nut.number1")))
let PhaseNumber = g.Key.key2
select new { Name, PhaseNumber, InnerD, Number, Weight };

//Gathering data for Nut2:
var nut2 = from n in Database.OfType<BoltGroup>()
//Making sure that nut2 exists in boltgroup:
where n.Bolt == true && n.Nut2 == true && n.GetTeklaReport("nut.weight2") > 0
//Grouping by name and phase:
group n by new { key1 = n.GetTeklaReport("nut.name2"), key2 = n.GetTeklaReport("phase") } into g
//Defining columns:
let Name = g.Key.key1
let InnerD = g.First().GetTeklaReport("nut.inner diameter2")
let Number = g.Sum(x => (int)x.GetTeklaReport("nut.number1"))
let Weight = g.Sum(x => (double)(x.GetTeklaReport("nut.weight2", 4)*x.GetTeklaReport("nut.number1")))
let PhaseNumber = g.Key.key2
select new { Name, PhaseNumber, InnerD, Number, Weight };

//Concatenating two lists (Hint: Lists must have same column names for concatenating):
var AllNuts = nut1.Concat(nut2);

//Walking through final nut list:
var result = from n in AllNuts
//Grouping by name and phase
group n by new { n.Name, n.InnerD, n.PhaseNumber } into g
//Defining required columns for final result:
let TotalNumber = g.Sum(x => x.Number)
let TotalWeight = Math.Round(g.Sum(x => (double)x.Weight), 4)
select new { g.Key.Name, g.Key.PhaseNumber, g.Key.InnerD, TotalNumber, TotalWeight };
//Passing the final result to the gridView:
return result;
