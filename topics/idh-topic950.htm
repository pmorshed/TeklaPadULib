// #############################################################
// ### Author: TeklaPad
// ### Desc.: Joining two picked polybeams.
// ### Compatibility: TS18 or newer
// ### Hint1: The order of selection matters.
// ### Hint2: Converting Tolist() allows  indexing ([0], [1],...)
// ### Hint3: In C#, An array index always starts from zero.
// #############################################################

Model m1 = new Model();
//Storing picked polybeams. Press the middle mouse button to terminate the selection.
var SelectedPolyBeams = m1.PickObjectsOfType<PolyBeam>();
//Making sure that there are only two PolyBeams in selection set:
if (SelectedPolyBeams != null && SelectedPolyBeams.Count() == 2)
{
	//According to Hint3, index 0 means first object:
	PolyBeam FirstPolyBeam = SelectedPolyBeams[0];
	//and index 1 means second object:
	PolyBeam SecondPolyBeam = SelectedPolyBeams[1];
	//Gathering points from second polybeam (Refer to Hint2):
	var points2 = SecondPolyBeam.Contour.ContourPoints.OfType<ContourPoint>().ToList();
	//Adding second polyline's points to first one:
	for (int i = 0; i < points2.Count(); i++)
	{   
		//Prevent from duplicate points:
		if (!FirstPolyBeam.Contour.ContourPoints.Contains(points2[i]))
		{
			FirstPolyBeam.AddContourPoint(points2[i]);
		}
	}
	SecondPolyBeam.Delete();
	FirstPolyBeam.Modify();
}
//Updating the model view:
m1.CommitChanges();
//Passing nothing to gridview:
return null;
