//## Author: TeklaPad
//## Description: Beam/PolyBeam joiner
//## Compatibility: TS18 or newer
//## Hint: The order of picking matters.
//## Hint: In C#, an array index always starts from zero.
//## Hint: The selection sequence is terminated by the middle button pick.


Model m1 = new Model();
//Picking:
var PickedBeams = m1.PickObjectsOfType<Part>();

PolyBeam JoindPolyBeam = new PolyBeam();
//Defining some properties according to first picked part:
//Hint: In C#, an array index always starts from zero
JoindPolyBeam.Profile.ProfileString = PickedBeams[0].Profile.ProfileString;
JoindPolyBeam.Material.MaterialString = PickedBeams[0].Material.MaterialString;
JoindPolyBeam.Position = PickedBeams[0].Position;
JoindPolyBeam.Class = PickedBeams[0].Class;

foreach (var item in PickedBeams)
{
    var points = item.GetContourPointList();
    foreach (var node in points)
    {
        JoindPolyBeam.AddContourPoint(node);
    }
    item.Delete();
}
JoindPolyBeam.Contour.ContourPoints = JoindPolyBeam.Contour.ContourPoints.RemoveDuplicates();
JoindPolyBeam.Insert();
m1.CommitChanges();
//Passing nothing to data grid:
return null;
