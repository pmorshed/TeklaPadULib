// ### Desc.: Converting solids to plates
// ### Compatibility: TS18 or newer

Model m1 = new Model();
//Walking through database (selected objects) and filtering parts:
foreach (var item in Database.OfType<Part>())
{
	var solid = item.GetSolid() as Solid;
	var faces = solid.GetFaceEnumerator().GetArray().OfType<Face>();
	//Walking through solid faces:
	foreach (var face in faces)
	{
		var loops = face.GetLoopEnumerator().GetArray().OfType<Loop>();
		//Walking through face loops:
		foreach (var loop in loops)
		{
			var vertices = loop.GetVertexEnumerator().GetArray().OfType<Point>();
			ContourPlate cp = new ContourPlate();
			//Walking through loop vertices:
			foreach (var point in vertices)
			{
				ContourPoint cpoint = new ContourPoint(point, null);
				cp.AddContourPoint(cpoint);
			}
			//Defining plate properties:
			cp.Profile.ProfileString = "PL10";
			cp.Class = "3";
			cp.Position.Depth = Position.DepthEnum.MIDDLE;
			//Inserting plate for each face:
			cp.Insert();
		}
	}
	//Removing original part:
	item.Delete();
}
//Updating the model view:
m1.CommitChanges();
//Passing nothing to gridview:
return null;
