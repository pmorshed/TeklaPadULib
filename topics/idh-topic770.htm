//Finding tapered rebar groups in selection:
var TaperedRebarGroups = from r in Database.OfType<RebarGroup>()
where r.GetTeklaReport("weight min") != r.GetTeklaReport("weight max")
select r;
//Ungrouping tapered rebar groups:        
foreach (var item in TaperedRebarGroups)
{
Operation.Ungrouping(item);
}
//Redraw the view:	        
new Model().CommitChanges();
//There is no need to pass anything to gridview. So:	        
return null;
