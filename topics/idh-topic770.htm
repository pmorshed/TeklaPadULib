//## Description: Create a spiral beam.
//## Compatibility: TS18 or newer


//General parameters:
var LocationX = 0;
var LocationY = 0;
var profile = "PL50*5";
var Height = 3000;

Model m1 = new Model();
PolyBeam pb = new PolyBeam();
pb.Profile.ProfileString = profile;
pb.Position.Depth = Position.DepthEnum.MIDDLE;
pb.Position.Plane = Position.PlaneEnum.MIDDLE;

for (int i = 0; i < 100; i++)
{
    double z = Height - ((Height / 100) * i);
    double angle = (Convert.ToDouble(i) % 20) * 18 / 180 * Math.PI;
    double dia = i * 10;

    double x = LocationX + dia * Math.Sin(angle);
    double y = LocationY + dia * Math.Cos(angle);
    ContourPoint p = new ContourPoint(new Tekla.Structures.Geometry3d.Point(x, y, z), null);
    pb.Contour.AddContourPoint(p);
}

pb.Insert();
//Updating model view:
m1.CommitChanges();
//Passing nothing to data grid:
return null;
