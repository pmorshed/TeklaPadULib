//## Description: Changing part color based on current color.
//## Compatibility: TS18 or newer
//## Hint: Make sure that your drawing is open.


var MyDrawing = new TSD.DrawingHandler().GetActiveDrawing();
foreach (var item in DrawingDatabase.OfType<TSD.Part>())
{
    if (item.Attributes.VisibleLines.Color == TSD.DrawingColors.Green)
    {
        item.Attributes.VisibleLines.Color = TSD.DrawingColors.Blue;
        item.Modify();
    }
}
//Updating the drawing view:
MyDrawing.CommitChanges();
//Passing nothing to the data grid:
return null;
