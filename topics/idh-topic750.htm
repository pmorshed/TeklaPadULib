//## Description: Assembly part list by pos
//## Compatibility: TS18 or newer
//## Hint: Make sure that numbering of objects is up to date.


//STEP-1
//Grouping assemblies by pos. number and selecting "Instance/No" pair from each group:
var MyAssemblyTable = from n in Database.OfType<Assembly>()
                    let Pos = n.GetTeklaReport("assembly pos")
                    group n by Pos into g
                    select new { MainAssemby = g.First(), MainAssemby_No = g.Count() };

//STEP-2:List of assembly parts:
//Walking through items inside assemblies of "AssemblyTable":                 
var AssemblyParts = from n in MyAssemblyTable
                    //Walking through parts inside each assembly:
                    from Part p in n.MainAssemby.PartList()
                    //Defining variables for grouping:
                    let key1 = p.GetTeklaReport("part pos") //part pos
                    let key2 = n.MainAssemby.GetTeklaReport("assembly pos") //main assembly pos
                    let key3 = n.MainAssemby_No
                    group p by new { key1, key2, key3 } into g
                    //Keeping one instance (first one) from each group of data:
                    let part = g.First()
                    //Defining required columns:
                    let AssemblyNo = g.Key.key3
                    let MainAssembly = g.Key.key2 + " [No:" + AssemblyNo + "]"
                    let Pos = g.Key.key1
                    let Type = "Part " + "[" + part.Profile.ProfileString + "]"
                    let No = g.Count()
                    let UnitWeight = part.GetTeklaReport("weight net")
                    let Weight_In_Assembly = g.Sum(x => (double)x.GetTeklaReport("weight net"))
                    let Weight_Tot = Weight_In_Assembly * AssemblyNo
                    select new { MainAssembly, Type, Pos, UnitWeight, No, Weight_In_Assembly, Weight_Tot };
//Passing the final result to the data grid:
return AssemblyParts;
