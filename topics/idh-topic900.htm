//## Author: TeklaPad
//## Description: Access to model objects from drawing objects.
//## Compatibility: TS18 or newer
//## Hint: Make sure that your drawing is open.


//Walking through selected drawing objects (DrawingDatabase) and getting collection of related model objects:
var result = from n in DrawingDatabase.GetModelObjectsOfType<Part>()
             //Defining required variables:
             let profile = n.Profile.ProfileString
             let weight = n.GetTeklaReport("weight net")
             let volume = n.GetTeklaReport("volume net")
             select new { profile, weight, volume };
//Passing the final result to the data grid:
return result;
