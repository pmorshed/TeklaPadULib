//## Description: Showing center lines only for some profiles.
//## Compatibility: TS18 or newer
//## Hint: Make sure that your drawing is open.


var MyDrawing = new TSD.DrawingHandler().GetActiveDrawing();
foreach (var item in DrawingDatabase.OfType<TSD.Part>())
{
    if (item.IsValidModelObjectOfType<Beam>())
    {
        var BeamInModel = item.GetModelObjectOfType<Beam>();
        //Condition (example: only for UNP profiles):
        if (BeamInModel.Profile.ProfileString.StartsWith("UNP"))
        {
            item.Attributes.DrawCenterLine = true;
        }
        else
        {
            item.Attributes.DrawCenterLine = false;
        }
        item.Modify();
    }
}
//Updating the drawing view:
MyDrawing.CommitChanges();
//Passing nothing to the data grid:
return null;
