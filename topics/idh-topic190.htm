//Walking through database items(selected items) and filtering "Parts":
var query = from n in Database.OfType<Part>()
//Defining required columns:
let id = n.Identifier.ID
let Profile = n.Profile.ProfileString
let length_mm = n.GetTeklaReport("length",0)
let Material = n.Material.MaterialString
//Converting to square meter and rounding after coversion:
let Area_m2 = Math.Round(n.GetTeklaReport("area net", 6) / 1000000, 3)
let Weight = n.GetTeklaReport("weight net")
//Selecting a new table (from orginal database) with required columns: 
select new { id, Profile, Material, length_mm, Area_m2, Weight};
//Passing the final result to the gridview:
return query;
