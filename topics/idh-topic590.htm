//Walking through database items(selected items) and filtering "Parts":
var result = from p in Database.OfType<Part>()
//Grouping by position number and phase:
let pos = p.GetTeklaReport("part pos")
let phase = p.GetTeklaReport("phase")
group p by new { pos, phase } into g
//defining required columns:
let Pos = g.Key.pos
let PartPhase = g.Key.phase
let No = g.Count()
let Profile = g.First().Profile.ProfileString
let Weight = g.First().GetTeklaReport("weight net", 3)
let length = g.First().GetTeklaReport("length net")
let TotalLength = length*No
let TotalWeight = Weight*No
//Selecting a new table (from orginal database) with required columns: 
select new {PartPhase, Pos, No, Profile, length, TotalLength, Weight, TotalWeight };
//Passing the final result to the gridview:
return result;
