//## Compatibility: TS18 or newer


//Walking through database items and filtering "Parts":
var query = from n in Database.OfType<Part>()
            //Grouping by profile:
            group n by n.Profile.ProfileString into g
            //Defining required columns:
            let WeightTotal = g.Sum(x => (double)x.GetTeklaReport("weight net"))
            //Rounding after summation:
            let LengthTotal = Math.Round(g.Sum(x => (double)x.GetTeklaReport("length net")), 2)
            let Profile = g.Key
            select new { Profile, LengthTotal, WeightTotal };
//Passing the final result to the data grid:
return query;
