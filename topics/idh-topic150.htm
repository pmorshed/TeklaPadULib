//Walking through database items(selected items):
var query = from n in Database
//Filtering "Parts" from the database:
let p = n as Part
where p != null
//Grouping by profile:
group p by p.Profile.ProfileString into g
//Defining required columns:
let WeightTotal = g.Sum(x => (double)x.GetTeklaReport("weight net"))
//Rounding after summation
let LengthTotal = Math.Round(g.Sum(x => (double)x.GetTeklaReport("length net")), 2)
let Profile = g.Key
//Selecting a new table (from orginal database) with required columns: 
select new { Profile, LengthTotal, WeightTotal };
//Passing the final result to the gridview:
return query;
