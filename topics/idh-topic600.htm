//## Author: TeklaPad
//## Description: Cast-unit part list by ID
//## Measurement units: Metric
//## Compatibility: TS18 or newer


//Walking through cast-unit:
var CUParts = from n in Database.OfType<Assembly>()
              //Walking through cast unit parts:
              from Part p in n.PartList()
              //Defining required variables:
              let Type = n.GetTeklaReport("cast unit type")
              let CU = n.Name + " [" + Type + "]" + " [ID:" + n.Identifier.ID + "]"
              let Part = p.Profile.ProfileString
              let Weight = p.GetTeklaReport("weight net")
              let Volume = Math.Round(p.GetTeklaReport("volume net", 9) / 1000000000, 3)
              select new { CU, Part, Weight, Volume };
//Passing the final result to the data grid
return CUParts;
