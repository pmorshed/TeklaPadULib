//Hint: washer1/washer2/washer3 may have different standards (depends on bolt assembly catalog)
	        
//Gathering data for Washers1:
var washer1 = from n in Database.OfType<BoltGroup>()
//Filtering washer1:
where n.Bolt==true && n.Washer1 == true
group n by n.GetTeklaReport("washer.name1") into g
let Name = g.Key
let InnerD = g.First().GetTeklaReport("washer.inner diameter1")
let Number = g.Sum(x => x.GetTeklaReport("washer.number1"))
let Weight = g.Sum(x => (double)(x.GetTeklaReport("washer.weight1", 4)*x.GetTeklaReport("washer.number1")))
select new { Name, InnerD, Number, Weight };

//Gathering data for Washers2:
var washer2 = from n in Database.OfType<BoltGroup>()
//Filtering washer2:
where n.Bolt==true && n.Washer2 == true
group n by n.GetTeklaReport("washer.name2") into g
let Name = g.Key
let InnerD = g.First().GetTeklaReport("washer.inner diameter2")
let Number = g.Sum(x => x.GetTeklaReport("washer.number2"))
let Weight = g.Sum(x => (double)(x.GetTeklaReport("washer.weight2", 4)*x.GetTeklaReport("washer.number2")))
select new { Name, InnerD, Number, Weight };

//Gathering data for Washers3:
var washer3 = from n in Database.OfType<BoltGroup>()
//Filtering washer3:
where n.Bolt==true && n.Washer3 == true
group n by n.GetTeklaReport("washer.name3") into g
let Name = g.Key
let InnerD = g.First().GetTeklaReport("washer.inner diameter3")
let Number = g.Sum(x => x.GetTeklaReport("washer.number3"))
let Weight = g.Sum(x => (double)(x.GetTeklaReport("washer.weight3", 4)*x.GetTeklaReport("washer.number3")))
select new { Name, InnerD, Number, Weight };

//Concatenating three lists (Hint: Lists must have same column names for concatenating):
var AllWashers = washer1.Concat(washer2).Concat(washer3);

var result = from n in AllWashers
group n by new { n.Name, n.InnerD } into g
let TotalNumber = g.Sum(x => x.Number)
let TotalWeight = Math.Round(g.Sum(x => (double)x.Weight), 4)
select new { g.Key.Name, g.Key.InnerD, TotalNumber, TotalWeight };
return result;
