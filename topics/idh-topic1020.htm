//## Description: Removing all cut lines from selected parts.
//## Compatibility: TS18 or newer


Model m1 = new Model();
foreach (var item in Database.OfType<Part>())
{
    foreach (var cut in item.GetBooleans().GetArray().OfType<CutPlane>())
    {
        cut.Delete();
    }
}
m1.CommitChanges();
