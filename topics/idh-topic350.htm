//Walking through database items(selected items) and filtering "Bolts":
var result = from n in Database.OfType<BoltGroup>()
//Removing holes:
where n.Bolt == true
//Grouping by name and type:
let key1 = n.GetTeklaReport("name")
let key2 = n.GetTeklaReport("site workshop")
let key3 = n.BoltStandard
group n by new { key1, key2, key3 } into g
//Defining required columns:
let Bolt = g.Key.key1
let Type = g.Key.key2
let AssemblyStandard = g.Key.key3 //Assembly cataloge standard
let No = g.Sum(x => x.GetTeklaReport("number"))
let TotalLength = g.Sum(x => (double)(x.GetTeklaReport("length") * x.GetTeklaReport("number")))
//Selecting a new table (from orginal database) with required columns:
select new { Bolt, AssemblyStandard, Type, No, TotalLength };
//Passing the final result to the gridview:
return result;
