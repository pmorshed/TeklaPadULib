//Walking through database items(selected items) and filtering "Assemblies":
var query = from n in Database.OfType<Assembly>()
//Removing cast units:
where n.GetTeklaReport("material type") == "STEEL" 
//Grouping by position number:
let Pos = n.GetTeklaReport("assembly pos")
group n by Pos into g
let FirstObjectOfGroup = g.First()
//Defining required columns:
let No = g.Count()
let MP = FirstObjectOfGroup.GetMainPart() as Part
let MainPart = MP.Profile.ProfileString
let Pos = g.Key // or Pos=FirstObjectOfGroup.GetTeklaReport("assembly pos")
let Weight = FirstObjectOfGroup.GetTeklaReport("weight net", 3)
let Name = FirstObjectOfGroup.Name
let WeightTotal = Weight * No
//Selecting a new table (from orginal database) with required columns:
select new { Pos, Name, No, MainPart, Weight, WeightTotal };
//Passing the final result to the gridview:
return query;
