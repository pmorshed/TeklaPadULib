//Hint: nut1 and nut2 may have different standards (depends on bolt assembly catalog)

//Step-1: Gathering data for Nuts1:
//Walking through database items(selected items) and filtering "Bolts":
var nut1 = from n in Database.OfType<BoltGroup>()
//Filtering nut1:
where n.Bolt == true && n.Nut1 == true && n.GetTeklaReport("nut.weight1") >0
//Defining required columns:
let Thickenes = n.GetTeklaReport("nut.thickness1")
let InnerD = n.GetTeklaReport("nut.inner diameter1")
let OuterD = n.GetTeklaReport("nut.outer diameter1")
let Name = n.GetTeklaReport("nut.name1")
let number = n.GetTeklaReport("nut.number1")
let weight = n.GetTeklaReport("nut.weight1")*number
let location="Nut1"
//Selecting a new table (from orginal database) with required columns:
select new { Name, Thickenes, InnerD, OuterD, number, weight, location, n.Identifier.GUID };

//Step-2: Gathering data for Nuts2:
var nut2 = from n in Database.OfType<BoltGroup>()
//Filtering nut2:
where n.Bolt == true && n.Nut2 == true && n.GetTeklaReport("nut.weight2") >0
//Defining required columns
let Thickenes = n.GetTeklaReport("nut.thickness2")
let InnerD = n.GetTeklaReport("nut.inner diameter2")
let OuterD = n.GetTeklaReport("nut.outer diameter2")
let Name = n.GetTeklaReport("nut.name2")
let number = n.GetTeklaReport("nut.number1")
let weight = n.GetTeklaReport("nut.weight2")*number
let location="Nut2"
//Selecting a new table (from orginal database) with required columns:
select new { Name, Thickenes, InnerD, OuterD, number, weight, location, n.Identifier.GUID };

//Step3: Concatenating two lists (Hint: Lists should have same column names for concatenating):
var result = nut1.Concat(nut2);
//Passing the final result to the gridview:
return result;
