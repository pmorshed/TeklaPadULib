// ### Desc.: Batch edit on drawing list's data (Name, Title1, Title2,...) 
// ### Compatibility: TS18 or newer
// ### Hint: Select some drawings from drawing list.
// ########################################################################

Model MyModel = new Model();
//Getting selected drawings (in drawing list):
var Drawings = MyModel.GetSelectedDrawings();
foreach (var item in Drawings)
{
	//Some conditions:
	if (item is TSD.AssemblyDrawing)
	{
		item.Title1 = item.Title1.Replace("ABC", "RR");
		item.Modify();
	}
}
MyModel.CommitChanges();
//Passing nothing to data table:
return null;
