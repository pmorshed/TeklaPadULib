//## Description: Converting all mark's text elements to uppercase.
//## Compatibility: TS18 or newer
//## Hint: Make sure that your drawing is open.


var MyDrawing = new TSD.DrawingHandler().GetActiveDrawing();
//Filtering "Mark" objects in selected drawing objects (DrawingDatabase):
var AllMarks = DrawingDatabase.OfType<TSD.Mark>();
//Walking through "Mark" collection
foreach (var mark in AllMarks)
{
    //Mark contents:
    var markContent = mark.Attributes.Content.GetEnumerator().GetArray();
    //Walking through each mark's text elements:
    foreach (var item in markContent.OfType<TSD.TextElement>())
    {
        //Converting to uppercase:
        item.Value = item.Value.ToUpper();
    }
    mark.Modify();
}
//Updating the drawing view:
MyDrawing.CommitChanges();
//Passing nothing to the data grid:
return null;
