//Walking through database items(selected items) and filtering "Assemblies":
var query = from n in Database.OfType<Assembly>()
//Filtering cast units:
where n.GetTeklaReport("material type") == "CONCRETE"
//Grouping by cast unit position number:
let CU_POS = n.GetTeklaReport("cast unit pos")
group n by CU_POS into g
//Defining required columns:
let Pos = g.Key
let No = g.Count()
let Name = g.First().GetTeklaReport("cast unit name")
let Material = g.First().GetTeklaReport("material")
let RebarWeightInCU = g.First().GetTeklaReport("cast unit rebar weight", 4)
let CU_Type = g.First().GetTeklaReport("cast unit type")
//Converting from from mm3 to m3:
let CU_Volume = Math.Round(g.First().GetTeklaReport("volume net", 9) / 1000000000, 3)
//Converting from Kg to Ton:
let CU_Weight = g.First().GetTeklaReport("weight", 4) / 1000
let Vol_Tot = CU_Volume * No
let W_Tot = CU_Weight * No
//Selecting a new table (from orginal database) with required columns:
select new { Pos, No, Name, Material, CU_Type, RebarWeightInCU, CU_Volume, CU_Weight, Vol_Tot, W_Tot };
//Passing the final result to the gridview:
return query;
