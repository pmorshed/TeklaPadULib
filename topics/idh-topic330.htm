//## Description: Bolt list with GUID
//## Compatibility: TS18 or newer


//Walking through database items(selected items) and filtering "Bolts":
var result = from n in Database.OfType<BoltGroup>()
             //Removing holes:
             where n.Bolt == true
             //Defining requiered columns:
             let type = n.GetTeklaReport("site workshop")
             let No = n.GetTeklaReport("number")
             let Standard = n.BoltStandard //Bolt assembly catalog standard
             let GUID = n.Identifier.GUID
             //Hint: "NAME" attribute contains [size]-[length]-[bolt standard]
             let Name = n.GetTeklaReport("name")
             let ConnectedParts = n.GetTeklaReport("connected parts")
             let phase = n.GetTeklaReport("phase")
             select new { Name, Standard, No, type, phase, ConnectedParts, GUID };
//Passing the final result to the data grid:
return result;
