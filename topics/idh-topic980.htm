//## Description: Creating an elliptical contour plate.
//## Compatibility: TS18 or newer


Model m1 = new Model();
ContourPlate cplate = new ContourPlate();
var R1 = 3000;
var R2 = 1500;
for (int i = 0; i < 360; i += 10)
{
    var angle = (Math.PI * i) / 180;
    var x = R1 * Math.Cos(angle);
    var y = R2 * Math.Sin(angle);
    Point p = new Point(x, y, 0);
    ContourPoint cp = new ContourPoint(p, null);
    cplate.Contour.AddContourPoint(cp);
}
cplate.Profile.ProfileString = "PL20";
cplate.Insert();
//Updating the model view:
m1.CommitChanges();
//Passing nothing to data grid:
return null;
