// ### Desc.: Converting all selected "Texts" to uppercase
// ### Compatibility: TS18 or newer
// ### Hint: Make sure that your drawing is open.
// #############################################################

var MyDrawing = new TSD.DrawingHandler().GetActiveDrawing();

//Filtering "Text" objects in DrawingDatabase items:
var AllTexts = DrawingDatabase.OfType<TSD.Text>();
//Walking through created collection:
foreach (var item in AllTexts)
{
//Converting each item to uppercase:
item.TextString = item.TextString.ToUpper();
item.Modify();
}
//Updating the drawing view:
MyDrawing.CommitChanges();
//Passing nothing to the gridview:
return null;
