//## Description: Converting PolyBeams to contour plates
//## Compatibility: TS18 or newer


Model m1 = new Model();
//Walking through database items(selected items) and filtering polybeams:
foreach (var item in Database.OfType<PolyBeam>())
{
    //Creating contour plate:
    ContourPlate cplate = new ContourPlate();
    cplate.Profile.ProfileString = "PL50";
    cplate.Contour.ContourPoints = item.Contour.ContourPoints;
    //Removing the polybeam:
    item.Delete();
    cplate.Insert();
}
//Updating the model view:
m1.CommitChanges();
//Passing nothing to data grid:
return null;
