//Walking through database items(selected items):
var result = from n in Database
//Filtering "Bolts" from the database:
let b = n as BoltGroup
where b != null
//Removing holes:
where b.Bolt == true
//Grouping by size:
group b by b.BoltSize  into g
//Defining required columns:
let BoltSize = g.Key
let No = g.Sum(x => x.GetTeklaReport("number"))
let TotalLength = g.Sum(x => (double)(x.GetTeklaReport("length") * x.GetTeklaReport("number")))
//Selecting a new table (from orginal database) with required columns:
select new { BoltSize, No, TotalLength };
//Passing the final result to the gridview:
return result;
