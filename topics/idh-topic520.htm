// ### Desc.: Cast unit rebars with pull-out picture by pos
// ### Compatibility: TS18 or newer
// ### Hint: Make sure that numbering of objects is up to date.
// #############################################################

//Walking through database items(selected items) and filtering "Assemblies":
var result = from a in Database.OfType<Assembly>()
//Filtering cast units:
where a.GetTeklaReport("material type") == "CONCRETE"
from Part p in a.PartList()
from Reinforcement r in p.GetReinforcements().GetArray()
//Removing strands:
where r.GetType().Name != "RebarStrand" && r.IsGeometryValid()
//Grouping by rebar pos:
group r by new { key1 = r.GetTeklaReport("rebar pos"), key2 = r.GetTeklaReport("grade") }  into g
let rebar = g.First()
//-------Temporary variable(Returns "true" for tapered groupes-----
let IsTapered = rebar.GetTeklaReport("length min") != rebar.GetTeklaReport("length max")
//-----------------------------------------------------------------
//Defining required columns:
let Size = rebar.GetTeklaReport("size")
let Pos = g.Key.key1
let Grade = g.Key.key2
let Shape = rebar.GetTeklaReport("shape")
let Pcs = g.Sum(x => x.GetNumberOfRebars()) //or g.Sum(x => x.GetTeklaReport("number"))
let Unitlength = IsTapered==true ? "VAR" : rebar.GetTeklaReport("length")
let UnitWeight = IsTapered==true ? "VAR" : rebar.GetTeklaReport("weight")
//Hint: It's recommended to use "WEIGHT_TOTAL_IN_GROUP" attribute instead of "WEIGHT_TOTAL" in newer versions of Tekla
let TotalWeight = IsTapered==true ? g.Sum(x => (double)x.GetTeklaReport("weight total", 6)) : UnitWeight*Pcs
let TotalLength = IsTapered? Math.Round(TotalWeight*rebar.GetTeklaReport("length",7)/rebar.GetTeklaReport("weight",7),3) : Unitlength*Pcs
//Hint: Make sure that rebar numbers are up to date while using "GetTeklaPullOut" method.
let pic = rebar.GetTeklaPullOut()
//Selecting a new table (from orginal database) with required columns:
select new { Pos, Size, Grade, Shape, Pcs, Unitlength, TotalLength, UnitWeight, TotalWeight, pic };
//Passing the final result to the gridview:
return result;
