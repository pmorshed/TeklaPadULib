//Walking through database items(selected items):
var query = from n in Database
//Filtering "Parts" from the database:
let p = n as Part
where p != null
//Defining required columns:
let PhaseNumber = p.GetTeklaReport("phase")
let Profile = p.Profile.ProfileString
let Material = p.Material.MaterialString
let Length = p.GetTeklaReport("length net", 0)
let Weight = p.GetTeklaReport("weight net")
//Selecting a new table (from orginal database) with required columns:
select new { PhaseNumber, Profile, Material, Length, Weight };
//Passing the final result to the gridview:
return query;
