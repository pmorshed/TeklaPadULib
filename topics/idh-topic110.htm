//## Description: Distinct part list by profile (by phase)
//## Compatibility: TS18 or newer


//Walking through database items and filtering "Parts":
var query = from n in Database.OfType<Part>()
            //Defining required columns:
            let PhaseNumber = n.GetTeklaReport("phase")
            let Profile = n.Profile.ProfileString
            let Material = n.Material.MaterialString
            let Length = n.GetTeklaReport("length net", 0) //Precision:0
            let Weight = n.GetTeklaReport("weight net")
            select new { PhaseNumber, Profile, Material, Length, Weight };
//Passing the final result to the data grid:
return query;
