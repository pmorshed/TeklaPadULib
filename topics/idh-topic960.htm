// ### Author: TeklaPad
// ### Desc.: Converting picked beams to polybeam
// ### Compatibility: TS18 or newer
// ### Hint1: Picking order affects the result.
// ### Hint2: Swapping the beam handles can affect the result. 
// #############################################################

Model m1 = new Model();
//Storing picked beams. Press the middle mouse button to terminate the selection.
var PickedBeams = m1.PickObjectsOfType<Beam>();
PolyBeam pb = new PolyBeam();
foreach (var item in PickedBeams)
{
	ContourPoint cp1 = new ContourPoint(item.StartPoint, null);
	ContourPoint cp2 = new ContourPoint(item.EndPoint, null);
	//Prevent from duplicate points:
	if (!pb.Contour.ContourPoints.Contains(cp1))
	{
		pb.AddContourPoint(cp1);
	}
	//Prevent from duplicate points:
	if (!pb.Contour.ContourPoints.Contains(cp2))
	{
		pb.AddContourPoint(cp2);
	}
	item.Delete();
}
pb.Profile.ProfileString = PickedBeams[0].Profile.ProfileString;
pb.Class = PickedBeams[0].Class;
pb.Insert();
m1.CommitChanges();
//Passing nothing to data table:
return null;
